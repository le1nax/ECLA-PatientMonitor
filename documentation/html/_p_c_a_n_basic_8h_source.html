332// Represents a PCAN message
333//
334typedef struct tagTPCANMsg
335{
336 DWORD ID; // 11/29-bit message identifier
337 TPCANMessageType MSGTYPE; // Type of the message
338 BYTE LEN; // Data Length Code of the message (0..8)
339 BYTE DATA[8]; // Data of the message (DATA[0]..DATA[7])
340} TPCANMsg;

342// Represents a timestamp of a received PCAN message
343// Total Microseconds = micros + (1000ULL * millis) + (0x100000000ULL * 1000ULL * millis_overflow)
344//
345typedef struct tagTPCANTimestamp
346{
347 DWORD millis; // Base-value: milliseconds: 0.. 2^32-1
348 WORD millis_overflow; // Roll-arounds of millis
349 WORD micros; // Microseconds: 0..999
350} TPCANTimestamp;

358 BYTE DLC; // Data Length Code of the message (0..15)
359 BYTE DATA[64]; // Data of the message (DATA[0]..DATA[63])
360} TPCANMsgFD;
